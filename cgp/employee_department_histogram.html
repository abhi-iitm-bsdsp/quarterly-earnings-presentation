import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Dataset
data = {'employee_id': ['EMP001', 'EMP002', 'EMP003', 'EMP004', 'EMP005'], 'department': ['R&D', 'HR', 'HR', 'Finance', 'Operations'], 'region': ['Middle East', 'Europe', 'Asia Pacific', 'North America', 'Asia Pacific'], 'performance_score': [83.91, 76.85, 76.48, 74.44, 66.71], 'years_experience': [5, 11, 15, 8, 14], 'satisfaction_rating': [4, 4.6, 4.5, 3.1, 4.4]}

df = pd.DataFrame(data)

# Count Finance employees
finance_count = df[df['department']=="Finance"].shape[0]
print("Number of employees in Finance Department:", finance_count)

# Histogram
sns.set_style("whitegrid")
plt.figure(figsize=(8,8))
sns.countplot(data=df, x='department', palette="Set2")
plt.title("Employee Count by Department")
plt.xlabel("Department")
plt.ylabel("Number of Employees")
plt.savefig("department_histogram.png", dpi=64, bbox_inches='tight')
plt.show()
