# employee_visualization.py
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Email for validation
email = "23f1001785@ds.study.iitm.ac.in"

# --- Generate synthetic dataset ---
np.random.seed(42)
departments = ['R&D', 'HR', 'Finance', 'Operations', 'Marketing', 'Sales', 'IT']
regions = ['North America', 'Europe', 'Asia Pacific', 'Middle East', 'South America']

employee_ids = [f"EMP{i:03d}" for i in range(1, 101)]
department_choices = np.random.choice(departments, size=100)
region_choices = np.random.choice(regions, size=100)
performance_scores = np.round(np.random.normal(loc=75, scale=10, size=100), 2)
years_experience = np.random.randint(1, 21, size=100)
satisfaction_ratings = np.round(np.random.uniform(3.0, 5.0, size=100), 1)

df = pd.DataFrame({
    "employee_id": employee_ids,
    "department": department_choices,
    "region": region_choices,
    "performance_score": performance_scores,
    "years_experience": years_experience,
    "satisfaction_rating": satisfaction_ratings
})

# --- Frequency count for Finance ---
finance_count = df[df['department'] == 'Finance'].shape[0]
print(f"Number of employees in Finance Department: {finance_count}")

# --- Seaborn Histogram ---
sns.set_style("whitegrid")
plt.figure(figsize=(8,8))  # 512x512 px
sns.countplot(data=df, x='department', palette="Set2")
plt.title(f"Employee Count by Department ({email})", fontsize=16)
plt.xlabel("Department")
plt.ylabel("Number of Employees")
plt.savefig("department_histogram.png", dpi=64, bbox_inches='tight')
plt.show()

# --- Save HTML including Python code and output ---
html_content = f"""

Output:


Number of employees in Finance Department: {finance_count}



Generated Histogram:




"""

with open("employee_department_histogram.html", "w") as f:
    f.write(html_content)
